Consultas SQL

Código PHP + MySQL que se ejecutaría en el servidor

INSERT Insertar un usuario (CREATE)
Añadir registro nuevo 
// Conexión a la base de datos con PDO (más seguro que mysqli)
$pdo = new PDO("mysql:host=localhost;dbname=mi_bbdd", $usuario, $password);
$pdo->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);

// Usamos prepared statements para evitar SQL Injection
$sql = "INSERT INTO usuarios (nombre, email, rol) VALUES (:nombre, :email, :rol)";
$stmt = $pdo->prepare($sql);

// Ejecutamos con los valores reales (nunca concatenamos directamente)
$stmt->execute([
    ':nombre' => $_POST['nombre'],
    ':email'  => $_POST['email'],
    ':rol'    => $_POST['rol']
]);
SELECT Obtener todos los usuarios (READ)
Leer registros 
// SELECT * trae todas las columnas; ORDER BY las ordena por fecha
$stmt = $pdo->query("SELECT * FROM usuarios ORDER BY creado_en DESC");

// fetchAll devuelve todos los resultados como array asociativo
$usuarios = $stmt->fetchAll(PDO::FETCH_ASSOC);

// Recorremos con foreach para mostrarlos en HTML
foreach ($usuarios as $usuario) {
    echo "<tr><td>" . htmlspecialchars($usuario['nombre']) . "</td></tr>";
    // htmlspecialchars evita XSS al mostrar datos en HTML
}
UPDATE Actualizar un usuario (UPDATE)
Modificar registro existente 
// Actualizamos solo las columnas que queremos cambiar
// WHERE es CRÍTICO: sin él se actualizarían TODOS los registros
$sql = "UPDATE usuarios SET nombre = :nombre, email = :email WHERE id = :id";
$stmt = $pdo->prepare($sql);

$stmt->execute([
    ':nombre' => $_POST['nombre'],
    ':email'  => $_POST['email'],
    ':id'     => $_POST['id']   // ID del usuario a modificar
]);
DELETE Eliminar un usuario (DELETE)
Borrar registro 
// DELETE también necesita WHERE obligatoriamente
// En producción se suele hacer "soft delete": marcar como inactivo en lugar de borrar
$sql = "DELETE FROM usuarios WHERE id = :id";
$stmt = $pdo->prepare($sql);

$stmt->execute([':id' => $_GET['id']]);

// Redirigimos de vuelta a la lista tras borrar
header("Location: usuarios.php?msg=eliminado");
exit;
